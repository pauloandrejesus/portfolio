Atualização = 
FORMAT (
    MAX ( 'TabAtualização'[Data da Última Atualização] ),
    "ddd dd/mm/yyyy",
    "PT-BR"
)

Divisor = 
SELECTEDVALUE ( TabUnidade[Divisor] )

---[ Estatística Descritiva ]--------------------------

Soma = 
VAR vValor =
    SUM ( TabLivros[Exemplares] )
RETURN
    DIVIDE ( vValor, [Divisor] )

Média = 
VAR vValor =
    AVERAGE ( 'TabLivros'[Exemplares] )
RETURN
    DIVIDE ( vValor, [Divisor] )

Mínimo =

    MIN ( 'TabLivros'[Exemplares] )
RETURN
    DIVIDE ( vValor, [Divisor] )

Máximo =

    MAX ( 'TabLivros'[Exemplares] )

    DIVIDE ( vValor, [Divisor] )


---[ Grandes Números ]--------------------------

∑ Exemplares = 
CALCULATE ( [Exemplares], ALLEXCEPT ( TabLivros, TabLivros[Ano] ) )

[Soma]

DIVIDE ( [Exemplares], [∑ Exemplares] )
